Gets a transaction from a block hash and transaction hash

#### Request

```js
{
  blockHash: string;
  transactionHash: string;
}
```

#### Response

```js
{
  fee: string
  expiration: number
  noteSize: number
  notesCount: number
  spendsCount: number
  signature: string
  notes: Array<{
    hash: string
    serialized: string
  }>
  spends: Array<{
    nullifier: string
    commitment: string
    size: number
  }>
  mints: Array<{
    assetId: string
    value: string
    name: string
    metadata: string
  }>
  burns: Array<{
    assetId: string
    value: string
  }>
  notesEncrypted: string[] // Deprecated: use `notes` instead
}
```
